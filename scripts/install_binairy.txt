$ apt update && sudo apt upgrade -y

# 1. DOWNLOAD + CHECK BINAIRY (CADDY + SCOPECACHE MODULE)

$ cd /tmp
$ wget https://github.com/VeloxCoding/scopecache/releases/download/v0.8.18/caddyscope-linux-amd64
$ wget https://github.com/VeloxCoding/scopecache/releases/download/v0.8.18/SHA256SUMS
$ sha256sum --check --ignore-missing SHA256SUMS

# caddyscope-linux-amd64: OK

# 2. MOVE AND CHANGE SYSTEM USER

$ mv /tmp/caddyscope-linux-amd64 /usr/local/bin/caddy
$ chmod +x /usr/local/bin/caddy

$ caddy version

# v2.11.2 h1:iOlpsSiSKqEW+SIXrcZsZ/NO74SzB/ycqqvAIEfIm64=

# 3. CREATE CADDY GROUP

$ groupadd --system caddy

$ useradd --system \
    --gid caddy \
    --create-home \
    --home-dir /var/lib/caddy \
    --shell /usr/sbin/nologin \
    --comment "Caddy web server" \
    caddy

# 4. CREATE CADDY FILE

$ mkdir -p /etc/caddy

$ tee /etc/caddy/Caddyfile > /dev/null <<'EOF'
{
    admin off
}

:80 {
    scopecache {
        scope_max_items 100000
        max_store_mb    100
        max_item_mb     1
    }
    respond 404
}
EOF

# check file:
$ cat /etc/caddy/Caddyfile


# 5. RUN CADDY AS SYSTEM SERVICE (restarts at crash)

$ tee /etc/systemd/system/caddy.service > /dev/null <<'EOF'
[Unit]
Description=Caddy with scopecache module
Documentation=https://caddyserver.com/docs/
After=network.target network-online.target
Requires=network-online.target

[Service]
Type=notify
User=caddy
Group=caddy
ExecStart=/usr/local/bin/caddy run --environ --config /etc/caddy/Caddyfile
ExecReload=/usr/local/bin/caddy reload --config /etc/caddy/Caddyfile --force
TimeoutStopSec=5s
LimitNOFILE=1048576
PrivateTmp=true
ProtectSystem=full
AmbientCapabilities=CAP_NET_BIND_SERVICE
Restart=on-failure
RestartSec=5s

[Install]
WantedBy=multi-user.target
EOF

$ systemctl daemon-reload

$ systemctl enable --now caddy

# show status
$ systemctl status caddy --no-pager


# 6. CHECK IF CADDY + SCOPECACHE MODULE RUNS

$ curl http://localhost/help

# see instructions at https://github.com/VeloxCoding/scopecache/blob/main/docs/scopecache-core-rfc.md

$ curl -X POST http://localhost/append \
  -H 'Content-Type: application/json' \
  -d '{"scope":"demo","payload":{"msg":"hello ScopeCache"}}'

# response 
# {"ok":true,"item":{"scope":"demo","seq":1,"ts":1778308461412130},"duration_us":88}

$ curl http://localhost/get?scope=demo&seq=1
$ curl http://localhost/tail?scope=demo
$ <external-ip>/tail?scope=demo


# 7. INSTALL WRK (benchmark tool)

$ apt install -y wrk
